<script src="/static/intro_accessible.js"></script>

<script>
  $(document).ready(function(){
    
    /*****************************************************
    * This section enables the intro.js text blocks.
    * They're set in the addNavData() function below.
    *****************************************************/

    addNavData();
    
    $('#navmaker').on('click tap', function(event){
        event.preventDefault();
        introJs().start();
    });
    
    // End of intro.js code.
    
  });
  
  /**************************************************************
  * This function adds the data needed for the intro.js text blocks.
  * You can change the text if you want.
  **************************************************************/
  function addNavData(){
    let navmaker = $('#navmaker');
    navmaker.attr("data-step", "1");
    navmaker.attr("data-intro", "Welcome to our course! You are on the Course Outline page. Click on the 'next' button or press the right arrow key to continue. Press 'escape' or click 'done' to quit this tour.");
    navmaker.attr("data-position", "right");

    let courseware = $('.course-tabs .active');
    courseware.attr("data-step", "2");
    courseware.attr("data-intro", "The course outline, including pages with all the videos, problems, and more, can all be found in the Course link.");
    courseware.attr("data-position", "right");

    let discussion = $('.course-tabs').find('a:contains("Discussion")');
    discussion.attr("data-step", "3");
    discussion.attr("data-intro", "The Discussion link sends you to the discussion boards, where you can talk to your fellow students.");
    discussion.attr("data-position", "right");

    let progress = $('.course-tabs').find('a:contains("Progress")');
    progress.attr("data-step", "4");
    progress.attr("data-intro", "The Progress tab is where you can find your grade.");
    progress.attr("data-position", "right");

    let syllabus = $('.course-tabs').find('a:contains("Syllabus")');
    syllabus.attr("data-step", "5");
    syllabus.attr("data-intro", "The Syllabus tab will tell you about the course contents and policies. You will want to read this early.");
    syllabus.attr("data-position", "right");

    let outline = $('.course-outline > ol').find(">:first-child");
    outline.attr("data-step", "6");
    outline.attr("data-intro", "Finally, this clickable course outline will help you navigate through our material. Click the Done button or press Escape and you're ready to go!");
    outline.attr("data-position", "top");

  }
</script>
